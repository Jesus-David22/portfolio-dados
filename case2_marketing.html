# CASE 2 - OTIMIZAÇÃO DE CAMPANHA DE MARKETING (E-COMMERCE)
# ==================================================

# 1. Importar bibliotecas
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

print("=" * 50)
print("CASE 2: Otimização de Campanha de Marketing")
print("=" * 50)
print(f"✅ Pandas versão: {pd.__version__}")

# 2. Criar dataset de exemplo (campanhas de marketing)
print("\n📊 Criando dataset de campanhas...")
np.random.seed(42)

n_campanhas = 500

dados = pd.DataFrame({
    'campanha_id': range(1, n_campanhas + 1),
    'canal': np.random.choice(['Google Ads', 'Facebook Ads', 'Instagram', 'Email', 'TikTok'], n_campanhas),
    'produto': np.random.choice(['Camiseta', 'Calça', 'Tênis', 'Bolsa', 'Acessório'], n_campanhas),
    'investimento': np.random.uniform(100, 5000, n_campanhas),
    'cliques': np.random.poisson(500, n_campanhas),
    'conversoes': np.random.poisson(30, n_campanhas),
    'receita': np.random.uniform(500, 20000, n_campanhas)
})

# Adicionar margem de lucro (varia por produto)
margem_produto = {
    'Camiseta': 0.45, 'Calça': 0.50, 'Tênis': 0.40, 'Bolsa': 0.55, 'Acessório': 0.60
}
dados['margem'] = dados['produto'].map(margem_produto)

# Calcular métricas de performance
dados['ctr'] = (dados['conversoes'] / dados['cliques'] * 100).round(2)  # Click-Through Rate
dados['roas'] = (dados['receita'] / dados['investimento']).round(2)  # Return on Ad Spend
dados['lucro'] = dados['receita'] * dados['margem'] - dados['investimento']

print(f"✅ Dataset criado: {dados.shape[0]} campanhas, {dados.shape[1]} colunas")

# 3. Visualizar os dados
print("\n📋 Primeiras 10 linhas:")
print(dados.head(10))

# 4. Estatísticas gerais
print("\n📊 Estatísticas das campanhas:")
print(dados[['investimento', 'cliques', 'conversoes', 'receita', 'roas', 'lucro']].describe().round(2))

# 5. Análise por canal
print("\n🔍 Performance por Canal de Marketing:")
performance_canal = dados.groupby('canal').agg({
    'investimento': 'sum',
    'receita': 'sum',
    'lucro': 'sum',
    'roas': 'mean',
    'ctr': 'mean'
}).round(2)
performance_canal['roi'] = (performance_canal['lucro'] / performance_canal['investimento'] * 100).round(2)
print(performance_canal)

# 6. Análise por produto
print("\n🔍 Performance por Produto:")
performance_produto = dados.groupby('produto').agg({
    'investimento': 'sum',
    'receita': 'sum',
    'lucro': 'sum',
    'roas': 'mean',
    'margem': 'mean'
}).round(2)
print(performance_produto)

# 7. Gráficos
print("\n📈 Gerando gráficos de análise...")

fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# Gráfico 1: ROAS por Canal
canal_roas = dados.groupby('canal')['roas'].mean().sort_values(ascending=False)
canal_roas.plot(kind='bar', ax=axes[0, 0], color='skyblue', edgecolor='black')
axes[0, 0].set_title('ROAS Médio por Canal', fontsize=12, fontweight='bold')
axes[0, 0].set_ylabel('ROAS (Retorno sobre Investimento)')
axes[0, 0].axhline(y=2.0, color='red', linestyle='--', label='Meta ROAS=2.0')
axes[0, 0].legend()
axes[0, 0].tick_params(axis='x', rotation=45)

# Gráfico 2: Lucro vs Investimento
colors = ['green' if x > 0 else 'red' for x in dados['lucro']]
axes[0, 1].scatter(dados['investimento'], dados['lucro'], alpha=0.6, c=colors)
axes[0, 1].axhline(y=0, color='red', linestyle='--', alpha=0.5)
axes[0, 1].set_title('Lucro vs Investimento por Campanha', fontsize=12, fontweight='bold')
axes[0, 1].set_xlabel('Investimento (R$)')
axes[0, 1].set_ylabel('Lucro (R$)')

# Gráfico 3: ROAS por Produto
produto_roas = dados.groupby('produto')['roas'].mean().sort_values(ascending=False)
produto_roas.plot(kind='bar', ax=axes[1, 0], color='lightcoral', edgecolor='black')
axes[1, 0].set_title('ROAS Médio por Produto', fontsize=12, fontweight='bold')
axes[1, 0].set_ylabel('ROAS')
axes[1, 0].axhline(y=2.0, color='red', linestyle='--', label='Meta ROAS=2.0')
axes[1, 0].legend()
axes[1, 0].tick_params(axis='x', rotation=45)

# Gráfico 4: Margem vs ROAS
sns.scatterplot(data=dados, x='margem', y='roas', hue='produto', size='investimento', sizes=(20, 200), ax=axes[1, 1])
axes[1, 1].set_title('Margem de Lucro vs ROAS', fontsize=12, fontweight='bold')
axes[1, 1].set_xlabel('Margem de Lucro (%)')
axes[1, 1].set_ylabel('ROAS')
axes[1, 1].axhline(y=2.0, color='red', linestyle='--', alpha=0.5)

plt.tight_layout()
plt.show()

# 8. Clusterização de produtos (K-Means)
print("\n🤖 Aplicando Clusterização para segmentar produtos...")

# Preparar dados para clusterização
dados_cluster = dados.groupby('produto').agg({
    'roas': 'mean',
    'margem': 'mean',
    'investimento': 'sum'
}).reset_index()

# Normalizar dados
scaler = StandardScaler()
dados_normalizados = scaler.fit_transform(dados_cluster[['roas', 'margem', 'investimento']])

# Aplicar K-Means
kmeans = KMeans(n_clusters=3, random_state=42, n_init=10)
dados_cluster['cluster'] = kmeans.fit_predict(dados_normalizados)

# Mapear clusters para nomes
cluster_nomes = {
    0: '⭐ Estrelas (Alto ROAS + Alta Margem)',
    1: '💰 Vaca Leiteira (ROAS Médio + Margem Média)',
    2: '⚠️ Abacaxi (Baixo ROAS + Baixa Margem)'
}
dados_cluster['classificacao'] = dados_cluster['cluster'].map(cluster_nomes)

print("\n📊 Classificação dos produtos por cluster:")
print(dados_cluster[['produto', 'roas', 'margem', 'classificacao']].round(2))

# 9. Otimização de orçamento
print("\n" + "=" * 50)
print("OTIMIZAÇÃO DE ORÇAMENTO PROPOSTA")
print("=" * 50)

# Identificar campanhas com baixo ROAS (< 1.5)
baixo_roas = dados[dados['roas'] < 1.5]
alto_roas = dados[dados['roas'] > 3.0]

print(f"\n📉 Campanhas com ROAS < 1.5 (PÉSSIMO): {len(baixo_roas)} campanhas")
print(f"   Investimento total desperdiçado: R$ {baixo_roas['investimento'].sum():,.2f}")

print(f"\n📈 Campanhas com ROAS > 3.0 (EXCELENTE): {len(alto_roas)} campanhas")
print(f"   Investimento que deveria ser aumentado: R$ {alto_roas['investimento'].sum():,.2f}")

# Recomendação de realocação
economia = baixo_roas['investimento'].sum() * 0.7
realocacao = economia

print(f"\n💡 RECOMENDAÇÃO DE REALOCAÇÃO:")
print(f"   • Cortar 70% do orçamento das campanhas com ROAS < 1.5: R$ {economia:,.2f}")
print(f"   • Redistribuir esse valor para campanhas com ROAS > 3.0")

# Impacto projetado
impacto_projetado = realocacao * alto_roas['roas'].mean()
print(f"\n📈 IMPACTO PROJETADO:")
print(f"   • Novo investimento em campanhas de alto desempenho: R$ {realocacao:,.2f}")
print(f"   • Receita adicional projetada: R$ {impacto_projetado:,.2f}")
print(f"   • ROAS geral estimado após otimização: +{((impacto_projetado / dados['investimento'].sum()) * 100):.1f}%")

# 10. Conclusões
print("\n" + "=" * 50)
print("CONCLUSÕES DO CASE 2")
print("=" * 50)
print("""
1. Melhor canal de marketing: [ver gráfico] - maior ROAS
2. Pior canal de marketing: [ver gráfico] - menor ROAS
3. Produtos com maior margem: Acessório (60%) e Bolsa (55%)
4. Produtos com menor margem: Tênis (40%)
5. Clusterização identificou 3 grupos de produtos:
   - ESTRELAS: produtos com alto ROAS e alta margem → INVESTIR MAIS
   - VACA LEITEIRA: produtos estáveis → MANTER
   - ABACAXI: produtos com baixo desempenho → REDUZIR ou ELIMINAR

✅ RECOMENDAÇÕES PARA MARKETING:
- Realocar 70% do orçamento de campanhas com ROAS < 1.5
- Aumentar investimento em produtos classificados como "ESTRELAS"
- Considerar pausar canais com ROAS consistentemente abaixo de 1.5
- Criar campanhas específicas para produtos de alta margem
""")

print("\n✅ Case 2 concluído com sucesso!")
print("\n🎯 PORTFÓLIO COMPLETO!")
print("   Case 1: Análise de Churn (Previsão de Cancelamento)")
print("   Case 2: Otimização de Marketing (ROAS e Clusterização)")
